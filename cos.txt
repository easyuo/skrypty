initevents
set #sysmsgcol 1082
event sysmessage Skrypt [monitor_hp] odpalony (by Rada)
menu Clear
menu window title #CHARNAME
menu Window size 230 60
menu Window color black
menu Show
menu Font color Red
menu Font bgcolor black
menu Font size 20
menu Text reg010 5 0 HP
menu Font color Green
menu Text reg020 5 30 ST
menu Font color Gray
menu Text reg030 120 0 AR
menu Font color Yellow
menu Text reg040 120 30 WG
start:
reg1:
if %reg1 = #hits
goto reg2
menu Font color Red
set %reg1 #hits
menu delete reg1
menu Text reg1 60 0 %reg1
reg2:
if %reg2 = #stamina
goto reg3
menu Font color Green
set %reg2 #stamina
menu delete reg2
menu Text reg2 60 30 %reg2
reg3:
if %reg3 = #ar
goto reg4
menu Font color Gray
set %reg3 #ar
menu delete reg3
menu Text reg3 175 0 %reg3
reg4:
if %reg4 = #weight
goto start
menu Font color Yellow
set %reg4 #weight
menu delete reg4
menu Text reg4 175 30 %reg4
goto start
